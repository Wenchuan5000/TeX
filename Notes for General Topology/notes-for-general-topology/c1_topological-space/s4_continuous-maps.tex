\section{Continuous Maps}
%================================


%--------------------------------
\begin{definition}
	Let $(X, \mathcal T_X)$ and $(Y, \mathcal T_Y)$ be topological spaces. A function $f: X \to Y$ is \textit{continuous} iff
	$$
	\forall U \in \mathcal T_Y, \quad f^{-1}[U] \in \mathcal T_X.
	$$
\end{definition}
