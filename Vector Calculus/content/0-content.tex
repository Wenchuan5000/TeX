\chapter{Directional and Partial Derivatives}



\section{Basic Definitions}
%================================================
%ииииииииииииииииииииииииииииииииииииииииииииииии


%------------------------------------------------
\begin{definition}
	Let $f: U \to \mathbb R^n$, where $U$ is an open set in $\mathbb R^m$, and let $x \in U$.
	
	The \textit{$i$th partial derived function of $f$} is a function $\frac{\partial f}{\partial x_i}: U \to \mathbb R^n$
	$$
	\frac{\partial f}{\partial x_i}(x) :=
	\begin{cases}
		\displaystyle \lim_{\delta \to 0} \frac{f(x + \delta \hat e_i) - f(x)}{\delta} & \text{if the limit exists in $\mathbb R^n$}, \\
		0 & \text{otherwise}.
	\end{cases}
	$$
\end{definition}
